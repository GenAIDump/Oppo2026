def create_access_token(data: dict, expires_delta: Optional[timedelta] = None) -> str:
    to_encode = data.copy()
    if expires_delta:

    else:
        expire = datetime.now(timezone.utc) + timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES)

            provider: The LLM provider to use ("GEMINI", "OPENAI", "ANTHROPIC").

            model_name: Optional specific model name (e.g., "gemini-1.5-flash-latest", "gpt-4o"). Defaults set by provider.

            elif provider == "ANTHROPIC":

                    temperature=temperature,
                    messages=[{"role": "user", "content": prompt}]
                 )
                 if not response.content:

                 # Assuming response.content is a list of blocks, usually one TextBlock
                 raw_response_text = "".join(block.text for block in response.content if hasattr(block, 'text'))

            logger.debug(f"{provider} Raw Response: '{raw_response_text[:200]}...'")

            if expect_json:
                # Retry logic is handled by the decorator, but parsing might fail

        # --- Specific Exception Handling for each provider ---

        """

        # 1. Fetch all analysis results from MCP for the given post
        results: Dict[str, AnalysisResult] = {}
        try:
            # Assuming MCP endpoint like '/posts/{id}/analysis'
            mcp_response = await call_mcp_api(
                method="GET",
                endpoint=f"/posts/{internal_post_id}/analysis"
            )
            # Assuming response format is like: {"contradiction": {...}, "disinformation": {...}, ...}
            # Convert raw dicts back to AnalysisResult objects if needed
            for analysis_type, data in mcp_response.items():
                 if data: # Check if analysis exists
                    try:
                        results[analysis_type] = AnalysisResult(**data)
                    except Exception: # Handle potential Pydantic validation errors
                         logger.warning(f"Could not parse analysis result type '{analysis_type}' for post {internal_post_id}", exc_info=True)
            logger.debug(f"Fetched {len(results)} analysis results for post {internal_post_id}")
            if not results:
                logger.warning(f"No analysis results found for post {internal_post_id}. Decision Engine cannot proceed.")
                return

        except Exception as e:
            logger.error(f"Failed to fetch analysis results from MCP for post {internal_post_id}: {e}", exc_info=True)
            return # Cannot proceed without results

        disinfo_score = 0.0

        }
        total_weight = sum(weights.values()) # Should be 1.0

                method="PUT",

            logger.error(f"Failed to update scores via MCP for post {internal_post_id}: {e}", exc_info=True)

        # 4. Check Thresholds and Trigger Distribution
        trigger_distribution = False
        trigger_reasons = []

        if disinfo_score >= DISINFO_THRESHOLD:
            trigger_distribution = True
            trigger_reasons.append(f"Disinformation score ({disinfo_score:.2f}) met threshold ({DISINFO_THRESHOLD})")
        # Add other specific triggers if needed (e.g., high contradiction AND low fact-check)

        # Check overall distribution threshold based on the highest contributing score or combined logic
        should_distribute = significance_score >= DISTRIBUTION_THRESHOLD or disinfo_score >= DISTRIBUTION_THRESHOLD # Example logic

        if should_distribute:

        """

        alert_payload = {
            "internal_post_id": internal_post_id,

            "significance_score": significance,

                while running and time.monotonic() - wait_start < n:
